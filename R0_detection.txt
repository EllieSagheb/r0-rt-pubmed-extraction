    "Title: Parallel Multi-Agent Extraction Pipeline (Dual GPU)\n",
    "\n",
    "This notebook implements a high-throughput pipeline using two GPUs simultaneously.\n",

    "def load_permanent_model(model_id: str, gpu_id: int):\n",
    "    \"\"\"Loads a model to a specific GPU and keeps it there.\"\"\"\n",
    "    device = f\"cuda:{gpu_id}\"\n",
    "    print(f\"\\n[System] Loading {model_id} on {device}...\")\n",
    "    \n",
    "    bnb_config = BitsAndBytesConfig(\n",
    "        load_in_4bit=True,\n",
    "        bnb_4bit_compute_dtype=torch.float16,\n",
    "        bnb_4bit_quant_type=\"nf4\"\n",
    "    )\n",
    "    \n",
    "    tokenizer = AutoTokenizer.from_pretrained(model_id)\n",
    "    \n",
    "    # device_map={\"\": gpu_id} forces the entire model onto one specific card\n",
    "    model = AutoModelForCausalLM.from_pretrained(\n",
    "        model_id,\n",
    "        quantization_config=bnb_config,\n",
    "        device_map={\"\": gpu_id}, \n",
    "        trust_remote_code=True\n",
    "    )\n",
    "    \n",
    "    return model, tokenizer, device"
   ]

    "def agent_a_worker(input_queue, audit_queue, model, tokenizer, device):\n",
    "    \"\"\"\n",
    "    GPU 0 Worker: Reads JSON entries, extracts specific epidemiological data.\n",
    "    \"\"\"\n",
    "    print(f\"[Agent A] Ready on {device}. Processing abstracts...\")\n",
    "    \n",
    "    # SYSTEM PROMPT: Enforce the exact schema you asked for\n",
    "    system_prompt = \"\"\"You are an expert Epidemiologist. Extract the following fields from the abstract:\n",
    "    1. Disease Name (e.g., COVID-19, Monkeypox)\n",
    "    2. Location (e.g., Italy, Wuhan, Global)\n",
    "    3. Metric Type (R0 or Rt)\n",
    "    4. Value (The point estimate)\n",
    "    5. CI (The 95% Confidence Interval range, e.g., '1.2-1.5')\n",
    "    6. Evidence Snippet (Exact quote supporting the value)\n",
    "\n",
    "    Output strictly valid JSON:\n",
    "    {\n",
    "        \"disease\": \"string\",\n",
    "        \"location\": \"string\",\n",
    "        \"metric_type\": \"R0\" or \"Rt\",\n",
    "        \"value\": float,\n",
    "        \"ci\": \"string\",\n",
    "        \"evidence_snippet\": \"string\"\n",
    "    }\n",
    "    If multiple exist, extract the primary finding. If none, return empty JSON.\"\"\"\n",
    "\n",
    "    while True:\n",
    "        entry = input_queue.get()\n",
    "        \n",
    "        if entry is None:\n",
    "            audit_queue.put(None)\n",
    "            input_queue.task_done()\n",
    "            break\n",
    "            \n",
    "        # Get text and ID from your raw_abstracts.json\n",
    "        abstract_text = entry.get(\"text\", \"\")\n",
    "        article_id = entry.get(\"pmid\", \"Unknown\")\n",
    "        \n",
    "        # Prepare Prompt\n",
    "        messages = [\n",
    "            {\"role\": \"system\", \"content\": system_prompt},\n",
    "            {\"role\": \"user\", \"content\": f\"Abstract: {abstract_text}\"}\n",
    "        ]\n",
    "        text_input = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)\n",
    "        inputs = tokenizer(text_input, return_tensors=\"pt\").to(device)\n",
    "        \n",
    "        with torch.no_grad():\n",
    "            generated_ids = model.generate(**inputs, max_new_tokens=512, temperature=0.1)\n",
    "        \n",
    "        output_text = tokenizer.decode(generated_ids[0][len(inputs.input_ids[0]):], skip_special_tokens=True)\n",
    "        \n",
    "        try:\n",
    "            clean_json = output_text.replace(\"```json\", \"\").replace(\"```\", \"\").strip()\n",
    "            data = json.loads(clean_json)\n",
    "            \n",
    "            # Only proceed if we have a valid value\n",
    "            if data and data.get(\"value\"):\n",
    "                # Pass the extraction + the original Article ID to Agent B\n",
    "                payload = {\n",
    "                    \"article_id\": article_id,  # PRESERVE ID\n",
    "                    \"original_text\": abstract_text,\n",
    "                    \"extraction\": data\n",
    "                }\n",
    "                audit_queue.put(payload)\n",
    "            else:\n",
    "                pass \n",
    "\n",
    "        except Exception as e:\n",
    "            # print(f\"[Agent A] Error: {e}\")\n",
    "            pass\n",
    "            \n",
    "        input_queue.task_done()\n",
    "    print(\"\\n[Agent A] Shutting down.\")"

    "def agent_b_worker(audit_queue, final_database, model, tokenizer, device):\n",
    "    \"\"\"\n",
    "    GPU 1 Worker: Verifies R0, Rt, CI, and Location against the text.\n",
    "    \"\"\"\n",
    "    print(f\"[Agent B] Ready on {device}. Auditing...\")\n",
    "    \n",
    "    system_prompt = \"\"\"You are a QA Auditor. Verify the extracted data against the Abstract.\n",
    "    Check for:\n",
    "    1. Precision (Do the Value and CI match exactly?)\n",
    "    2. Context (Is the Location and Disease correct?)\n",
    "    3. Metric (Is it R0 or Rt?)\n",
    "    \n",
    "    Output JSON: {\"status\": \"PASS\" or \"FAIL\", \"reason\": \"string\"}\"\"\"\n",
    "    \n",
    "    while True:\n",
    "        payload = audit_queue.get()\n",
    "        \n",
    "        if payload is None:\n",
    "            audit_queue.task_done()\n",
    "            break\n",
    "            \n",
    "        # Unpack\n",
    "        article_id = payload[\"article_id\"]\n",
    "        extraction = payload[\"extraction\"]\n",
    "        original_text = payload[\"original_text\"]\n",
    "        \n",
    "        # Prepare Prompt\n",
    "        extraction_str = json.dumps(extraction, indent=2)\n",
    "        user_content = f\"Abstract: {original_text}\\n\\nExtracted Data:\\n{extraction_str}\"\n",
    "        \n",
    "        messages = [\n",
    "            {\"role\": \"system\", \"content\": system_prompt},\n",
    "            {\"role\": \"user\", \"content\": user_content}\n",
    "        ]\n",
    "        text_input = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)\n",
    "        inputs = tokenizer(text_input, return_tensors=\"pt\").to(device)\n",
    "        \n",
    "        with torch.no_grad():\n",
    "            generated_ids = model.generate(**inputs, max_new_tokens=256, temperature=0.0)\n",
    "            \n",
    "        output_text = tokenizer.decode(generated_ids[0][len(inputs.input_ids[0]):], skip_special_tokens=True)\n",
    "        \n",
    "        try:\n",
    "            clean_json = output_text.replace(\"```json\", \"\").replace(\"```\", \"\").strip()\n",
    "            critique = json.loads(clean_json)\n",
    "            \n",
    "            if critique.get(\"status\") == \"PASS\":\n",
    "                # Create the FINAL SCHEMA you requested\n",
    "                final_record = {\n",
    "                    \"article_id\": article_id,\n",
    "                    \"disease\": extraction.get(\"disease\"),\n",
    "                    \"location\": extraction.get(\"location\"),\n",
    "                    \"metric\": extraction.get(\"metric_type\"),\n",
    "                    \"value\": extraction.get(\"value\"),\n",
    "                    \"ci\": extraction.get(\"ci\"),\n",
    "                    \"evidence_snippet\": extraction.get(\"evidence_snippet\")\n",
    "                }\n",
    "                final_database.append(final_record)\n",
    "                print(f\"[PASS] ID: {article_id} | {final_record['disease']} {final_record['metric']}: {final_record['value']}\")\n",
    "                \n",
    "            else:\n",
    "                # print(f\"[FAIL] ID: {article_id} | Reason: {critique.get('reason')}\")\n",
    "                pass\n",
    "                \n",
    "        except:\n",
    "            print(\"[Agent B] Critical Error parsing critique JSON.\")\n",
    "            \n",
    "        audit_queue.task_done()\n",
    "    print(\"\\n[Agent B] Shutting down.\")"
